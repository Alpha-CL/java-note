# src.array

#### introduction

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


// src.array: 一组 "相同数据类型" 的数据( 创建数组时, 必须确认 "长度" & "数据类型" )


    1) 数组本是 "引用数据类型"
    
    2) 数组是在堆内存中的一串连续的 "地址"
    
    3) 数组初始化时必须指定长度
    
    4) 数组空间长度一旦确定, 不能再次发生改变
    
    5) 栈内存中存储的是数组的引用地址
    
    6) 数组内部存储的 任意类型( 原始值, 引用类型 )


//-------------------------------------------------------------------------------------------------------------------//


// 数组声明


[type] [] [array_name];             // 建议使用
[type] [array_name] [];


eg: 

    int [] x;
    char [] y;
    boolean [] z;


//-------------------------------------------------------------------------------------------------------------------//

// 数组赋值

//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//

 
// 静态初始化: 有长度, 有元素


[type] [] [array_name] = new [type] [] {child_01, child_02, child_03};

* 可以省略 "new [type] []", 但不建议


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


// 动态初始化: 有长度, 无元素

[type] [] [array_name] = new [type] [length] {};


当动态创建数组时, 子项默认值:

    0           // int
    0.0         // float
    0           // char
    false       // boolean
    null        // reference type


//-------------------------------------------------------------------------------------------------------------------//

// 访问数组子项

//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


// 通过数组索引访问数组子项( 索引 0 ~ +infinty )
// 数组长度: [0, length - 1]

int [] src.array = new int [] { 1, 2, 3, 4, 5, 6 }

init child_01 = src.array[0];
init child_02 = src.array[1];
init child_03 = src.array[2];
init child_04 = src.array[3];


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```

int [] arr = new int[] { 1, 2, 3, 4, 5, 6 };

#### enumerate

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


int [] arr = {1, 2, 3, 4, 5, 6};


// for遍历

for(int i = 0; i < arr.length; a++) {

    for(int j = 0; j = arr[i].length; j++) {
    
        System.out.println(arr[i][j]);
    }
}


// 增强遍历

for(int child:arr) {

    for(int value:child) {
    
        System.out.println(value);
    }
}


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```
