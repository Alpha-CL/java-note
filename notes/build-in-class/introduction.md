#### packaging class

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * 包装类      // jdk 1.5+: 自动拆装包 
 * 
 * 
 *  - 封包: 将 常量( 基本类型 ) 转换为 包装类( 引用类型 )
 *  - 解包: 将 包装类( 引用类型 ) 解构为 常量( 基本类型 )
 */


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


// 1.5-


Integer int = new Integer(10);              // 生成包装类

int intValue = int.intValue();              // 包装类解构 


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


// 1.5+

Integer int = 10;

// [type]Value: 属于重写的方法返回不同类型的基本类型的值

int intValue = new Integer(10);             // 自动拆解包


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```
